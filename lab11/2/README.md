2. Написання коду програми
```
#include <stdio.h>
#define size 3

int main(){
int mas[size][size];
int res[size*size] = {0};
int a = 0;
int *pRes = res;
for(int i = 0; i < size; i++){
	printf("\n");
	for(int j = 0; j < size; j++){
		*(*(mas+i)+j) = rand()%100;
		printf("%d ", mas[i][j]);
	}
}
printf("\n\n\n");

for(int i = 0; i < size; i++){
	for(int j = 0; j < size; j++){
		if( ((*(*(mas+i)+j))%2) == 0 && *(*(mas+i)+j) > 0){
		 pRes[a] = *(*(mas+i)+j);
		 printf("%d ", res[a]);
		 a++;
		}
	}
}

return 0;
}
```
